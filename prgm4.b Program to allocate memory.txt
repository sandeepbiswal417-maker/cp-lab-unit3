//prgm4.b Program to allocate memory using calloc() and display values
#include <stdio.h>
#include <stdlib.h>
int main() {
    int *ptr, n, i;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    ptr = (int *)calloc(n, sizeof(int));
    if (ptr == NULL) {
        printf("Memory allocation failed");
        return 1;
    }
    for (i = 0; i < n; i++) {
        printf("Enter value %d: ", i + 1);
        scanf("%d", &ptr[i]);
    }
    printf("Entered values:\n");
    for (i = 0; i < n; i++) {
        printf("%d ", ptr[i]);
    }

    return 0;
}
