//prgm 3c.c passing structure to function
#include <stdio.h>
struct student
{
    int id;
    char name[20];
    
};
void display(struct student *s)   
   {
       printf("ID: %d,name: %s\n\t\n",s->id,s->name);
   };
    int main()
    {
        struct student s1={101,"AAA"};
        struct student s2={102,"BBB"};
        
        display(&s1);
        display(&s2);
        return 0;
        
}
