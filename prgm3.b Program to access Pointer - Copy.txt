//prgm 3.b Program to access Pointer to structures
#include <stdio.h>
struct Student {
    int roll;
    char name[20];
    float marks;
};

int main() {
    struct Student s1 = {145, "Sandeep", 98.5};
    struct Student *ptr;

    ptr = &s1;

    printf("Roll: %d\n", ptr->roll);
    printf("Name: %s\n", ptr->name);
    printf("Marks: %.2f\n", ptr->marks);

    return 0;
}
